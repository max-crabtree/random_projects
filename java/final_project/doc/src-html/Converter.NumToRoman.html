398    /** Number to Roman Numeral Logic.
399     *
400     * This class holds the logic that converts a inputted number into a
401     * roman numeral. It takes in the input for the <a href =
402     * #getNum>getNum</a> method in the <a href =
403     * #Converter>Converter</a>class.
404     */
405    class NumToRoman {
406
407        /** Final numeral that is outputted. */
408        private StringBuffer total = new StringBuffer();
409
410        /** Length (in digits) of input number. */
411        protected int numLength;
412
413        /** The number of digits left while iterating. */
414        private int endLength;
415
416        /** Digit at current index. */
417        private int currentNum;
418
419        /** Current power (degree) of number at current index. */
420        private int currentPower;
421
422        /** Value of number at current index. */
423        private int powerValue;
424
425        /** Placeholder for each numeral iteration. */
426        private String numeral;
427
428        /** Initialise Variables. */
429        NumToRoman() {
430            total.setLength(0);
431            numLength = 0;
432            endLength = 0;
433            currentNum = 0;
434            currentPower = 0;
435            powerValue = 0;
436            numeral = "";
437        }
438
439        /** Processes user input and outputs results.
440         *
441         * This method processes the user input by using the <a href =
442         * #analyseNum>method</a> and generates an output.
443         *
444         * @return the final roman numeral String
445         */
446        protected String processNum() {
447
448            /* Reset total value for later rounds. */
449            total.setLength(0);
450            numLength = String.valueOf(num).length();
451
452            for (int i = 0; i < numLength; i++) {
453
454                analyseNum(numLength, i);
455
456                /* Special case, zero = N. */
457                if (num == 0) {
458                    total.append("N");
459                }
460
461                /* If there is nothing to add, skip. */
462                if (currentNum == 0) {
463                    continue;
464                }
465
466                /* If the number corresponds to a basic roman numeral... */
467                if (basic.containsKey(powerValue) == true) {
468                    basicRoman(powerValue);
469                }
470                else {
471                    complexRoman(powerValue, currentPower);
472                }
473
474                /* If we are testing... */
475                if (assertsEnabled) {
476
477                    /* Cosmetic Purposes. */
478                    if (powerValue == 0) {
479                        numeral = "-";
480                    }
481
482                    String diagnostics = "Current Number: %1d\tAmount: %2d\tRoman: %3s\tTotal: %4s";
483                    System.out.println(String.format(diagnostics, currentNum, currentNum * currentPower, numeral, total));
484                }
485            }
486            System.out.println("Roman Numeral is: " + total);
487            return total.toString();
488        }
489
490        /** Analyse user input and generate useful values for later
491         * calculations.
492         * 
493         * This method analyses the integer input and generates multiple
494         * values used for calculations in the <a href =
495         * #complexRoman>complexRoman</a> method. 
496         *
497         * @param length the total length of the number
498         * @param index the current index of the <code>length</code>
499         */
500        private void analyseNum(int length, int index) {
501            endLength = (length - index) - 1;
502            /* The '0' equates the char back to its numerical value (not
503             * ASCII). */
504            currentNum = String.valueOf(num).charAt(index) - '0';
505            currentPower = (int)Math.pow(10, endLength);
506            powerValue = currentNum * currentPower;
507        }
508
509        /** Basic roman numeral behaviour.
510         *
511         * This method checks the <code>basic</code> HashMap, and simply
512         * <code>append</code>s the respective value to the
513         * <code>total</code>.
514         *
515         * @param value the <code>powerValue</code> at the current
516         * <code>index</code>
517         */
518        private void basicRoman(int value) {
519            numeral = basic.get(value);
520            total.append(numeral);
521        }
522
523        /** More complex subtractive/additive behaviour.
524         *
525         * This method utilises many values from the <a href =
526         * #analyseNum>analyseNum</a> method. By using the roman numeral
527         * rules, the values are determined using a few conditional statements
528         * that hold reference to the degree of ten and how it relates to the
529         * 'basic' roman numerals.
530         *
531         * @param degreeValue the <code>powerValue</code> at the current index
532         * @param degree the degree of 10 at current index
533         */
534        private void complexRoman(int degreeValue, int degree) {
535            /* Always equates to first digit:
536             * eg. degreeValue = 3000 therefore degree = 1000
537             * therefore 3000 / 1000 = 3
538             */
539            int firstDigit = degreeValue / degree;
540
541            /* Subtractive Behaviour */
542            if (firstDigit == 4 || firstDigit == 9) {
543                numeral = subtract.get(degreeValue);
544                total.append(numeral);
545            }
546
547            /* Additive Behaviour */
548            else if (5 < firstDigit && firstDigit < 9)  {
549                /* If first digit is 6, 7 or 8 */
550                StringBuffer additive = new StringBuffer();
551                /* Take 5 from value so middle value is not repeated
552                 * example: 8000 -> 8 - 5 -> 3... so only append 3 values */
553                for (int i = 0; i < (firstDigit) - 5; i++) {
554                    additive.append(basic.get(degree));
555                }
556                /* Add additive value to middle value */
557                numeral = basic.get(5 * degree) + additive;
558                total.append(numeral);
559            }
560
561            /* Two and Three */
562            else {
563                /* Only runs if other conditionals don't go through. Appends
564                 * the basic degree numeral. */
565                StringBuffer twoAndThree = new StringBuffer();
566                for (int i = 0; i < (firstDigit); i++) {
567                    twoAndThree.append(basic.get(degree));
568                    /* If input is 2000 -> (i < (2000 / 1000)) -> append X
569                     * twice */
570                }
571                numeral = twoAndThree.toString();
572                total.append(numeral);
573            }
574        }
575    }

619         * By utilising the <a href = #rulesCheck>rulesCheck()</a> and <a href
620         * = #outputNum>outputNum()</a> methods, the roman numeral is

623        protected void processRoman() {
624            /* If the numeral is valid */
625            if (rulesCheck() == true) {

636         *
637         * By using the multiple rule checking methods:
638         *
639         *<a href = #checkNumeralValid>checkNumeralValid</a>
640         *<a href = #checkNumeralSize>checkNumeralSize</a>
641         *<a href = #checkNumeralAmount>checkNumeralAmount</a>
642         *
643         * The input is validated.
644         *
645         * @return <code>true</code> if number is valid. <code>false</code> if
646         * not
647         */
648        private boolean rulesCheck() {
649            checkNumeralValid();
650            if (isValid == true) {
651                checkNumeralSize();
652                checkNumeralAmount();
653            }
654            return isValid;
655        }

657        /** Check for mis-input.
658         *
659         * Checks if the character at the current index is contained within
660         * the <code>converted</code> HashMap. If it isn't,
661         * <code>isValid</code> is set to <code>false</code>.
662         */
663        private void checkNumeralValid() {
664            for (int i = 0; i < roman.length(); i++) {
665                current = Character.toString(roman.charAt(i));
666                if (converted.containsKey(current) == false) {
667                    System.err.println("[" + current + "] Invalid Numeral!");
668                    isValid = false;
669                }
670            }
671        }
672
673        /** Compare first and second char in pair.
674         *
675         * If the second numeral is greater in value than the first,
676         * <code>isValid</code> is set to <code>false</code>. The values for
677         * each of the characters are sourced from the <code>converter</code>
678         * HashMap.
679         * <p>
680         * The values compared exclude any values contained in the
681         * <code>s_converted</code> HashMap.
682         */
683        private void checkNumeralSize() {
684            for (int i = 0; i < roman.length(); i++) {
685                if (i + 2 <= roman.length()) {
686                    pair = roman.substring(i, i + 2);
687                    int first = converted.get(Character.toString(pair.charAt(0)));
688                    int second = converted.get(Character.toString(pair.charAt(1)));
689                    if (first < second && !(s_converted.containsKey(pair))) {
690                        System.err.println("[" + pair + "] Second number too large!");
691                        isValid = false;
692                        break;
693                    }
694                }
695            }
696        }
697
698        /** Compare groups of characters.
699         *
700         * If one unique numeral appears consectutively in a group of 4,
701         * <code>isValid</code> is set to <code>false</code>.
702         */
703        private void checkNumeralAmount() {
704            for (int i = 0; i < roman.length(); i++) {
705                if (i + 4 <= roman.length()) {
706                    quad = roman.substring(i, i + 4);
707                    char first = quad.charAt(0);
708                    char second = quad.charAt(1);
709                    char third = quad.charAt(2);
710                    char fourth = quad.charAt(3);
711                    if (first == second && second == third && third == fourth) {
712                        System.err.println("[" + quad + "] Too many numerals in a row!");
713                        isValid = false;
714                        break;
715                    }
716                }
717            }
718        }
719
720        /** Output number.
721         *
722         * This is run once the input is validated. It checks whether the
723         * <code>current</code> character is subtractive or basic, and invokes
724         * the relevant method.
725         *
726         * @return final number to be outputted
727         */
728        private int outputNum() {
729            for (int i = 0; i < roman.length(); i++) {
730
731                current = Character.toString(roman.charAt(i));
732
733                /* If current numeral is subtractive */
734                if (isSubtractive(i) == true) {
735                    complexNum(pair);
736                    i = i + 1;
737                }
738                /* These methods don't work. */
739                //else if (isOverbar(i) == true) {

744                //}
745                else { //or basic...
746                    basicNum(current);
747                }
748
749                result = number;
750
751                if (assertsEnabled) { //if testing...
752                    String diagnostics = "Current: %1s\tResult: %2d\tNumber: %3d\tPair: %4s\tQuad: %5s";
753                    System.out.println(String.format(diagnostics, current, result, number, pair, quad));
754                }
755            }
756            return number;
757        }

759        /** If numeral is subtractive.
760         *
761         * This runs through the input in pairs, and if it finds a
762         * 'subtractive' numeral, it <code>get</code>s the equivalent value
763         * from the <code>s_converted</code> HashMap.
764         *
765         * @param index current index of the character in the
766         * <code>roman</code> String
767         * @return <code>true</code> if the current number is subtractive,
768         * <code>false</code> if not
769         */
770        private boolean isSubtractive(int index) {
771            if (index + 2 <= roman.length()) {
772                pair = roman.substring(index, index + 2);
773                if (s_converted.containsKey(pair)) {
774                    return true;
775                }
776            }
777            return false;
778        }

803        /** Basic roman numeral process.
804         *
805         * Simply <code>get</code>s the roman numeral from the
806         * <code>converted</code> HashMap, and sets <code>number</code> equal
807         * to it plus <code>result</code>. The <code>result</code> variable
808         * keeps the previous value of the <code>number</code> variable, thus
809         * iteratively increasing the value each time until complete.
810         *
811         * @param basic the current basic numeral
812         */
813        private void basicNum(String basic) {
814            number = converted.get(basic) + result;
815        }

817        /** Subtractive roman numeral process.
818         *
819         * Very similar behaviour to the <a href = #basicNum>basicNum</a>
820         * method. The only difference being that the values are accessed from
821         * the <code>s_converted</code> HashMap.
822         *
823         * @param subtractive the current subtractive pair
824         */
825        private void complexNum(String subtractive) {
826            number = s_converted.get(subtractive) + result;
827        }

895/** Emulates the old Applet behaviour which is no longer supported.
896 *
897 * This class serves to provide the same sort of hooks (`init`, `start`,
898 * `paint`, `stop`, and `destroy`) that the Applet mechanism used to provide,
899 * while automatically handling all of the Frame construction, setup, and
900 * teardown. 
901 *
902 * @author Duane Leslie
903 */
904class FakeApplet extends Panel {
905    /** Automatically run after the constructor completes, but before any
906     * other method. */
907    public void init() {
908        System.err.println("init");
909    }
910    /** Called whenever the Applet is (re)started.
911     *
912     * This includes both initially, but also after a `stop()` call when the
913     * Applet is resumed (for example, after being minimized). */
914    public void start() {
915        System.err.println("start");
916    }
917    /** Called for every frame update to handle any custom drawing.
918     *
919     * It is important to note that 'widgets' automatically paint themselves
920     * and so don't need to be handled here. */
921    public void paint(Graphics g) {
922        super.paint(g);
923    }
924    /** Called whenever the Applet is stopped.
925     *
926     * This includes both when it is minimized, and also just before
927     * destruction when it is being closed. */
928    public void stop() {
929        System.err.println("stop");
930    }
931    /** Automatically run just before the Applet is destroyed. */
932    public void destroy() {
933        System.err.println("destroy");
934    }
935
936    /** Wait until the Event Dispatch Thread is empty before proceeding.
937     *
938     * This ensures that anything that is still trying to happen has happened
939     * before proceeding. This is most important for ensuring that the
940     * 'init()' routine is complete before trying to interact with the applet.
941     */
942    public static void sync() {
943        /* Ensure the EDT queue is empty and any 'invokeLater's are done */
944        System.err.println("waitEDT");
945        try {
946            EventQueue.invokeAndWait(new Runnable() { public void run() {}});
947        } catch (Exception e) {
948            e.printStackTrace();
949        }
950        System.err.println("emptyEDT");
951    }
952
953    /** Provides a mechanism to hold the main thread of the program until this
954     * Applet has closed.
955     *
956     * This doesn't need to be used for correct operation of an Applet
957     * program, but if you would like to retrieve some data once the Applet
958     * has closed (i.e. to save state or print out some statistics) in the
959     * main thread (instead of using the Applet's `destroy()` method) then you
960     * need to use this method to 'block' the main thread.
961     */
962    public synchronized void waitForClose() {
963        sync();
964
965        /* Check if we're still on the screen (i.e. can still notify) */
966        System.err.println("isDisplay");
967        if (!isDisplayable()) {
968            return;
969        }
970
971        /* Wait for notification */
972        System.err.println("waitNotify");
973        try {
974            this.wait();
975        } catch (Exception e) {
976            e.printStackTrace();
977        }
978        System.err.println("close");
979    }

981    /** Create a FakeApplet of size 600x300 pixels.
982     *
983     * When this constructor is called the Applet will immediately construct
984     * itself, and then on the Event Dispatch Thread it will run its `init()`
985     * method, display itself, and then run the `start()` method.
986     */
987    public FakeApplet() {
988        this(600, 300);
989    }

991    /** Create a FakeApplet of a custom size.
992     *
993     * Note that the size is the frame size, not the canvas size (which will
994     * be smaller due to decorations.
995     *
996     * @param width width of the frame in pixels
997     * @param height height of the frame in pixels
998     */
999    public FakeApplet(int width, int height) {
1000        super();
1001
1002        Frame frame = new Frame();
1003        frame.addWindowListener(new WindowAdapter() {
1004            public void windowOpened(WindowEvent e) {
1005                start();
1006            }
1007            public void windowIconified(WindowEvent e) {
1008                stop();
1009            }
1010            public void windowDeiconified(WindowEvent e) {
1011                start();
1012            }
1013            public void windowClosing(WindowEvent e) {
1014                stop();
1015                destroy();
1016                frame.dispose();
1017                synchronized (FakeApplet.this) {
1018                    FakeApplet.this.notify();
1019                }
1020            }
1021        });
1022        frame.setSize(width, height);
1023        frame.add(this);
1024
1025        /* Defer init and display until the constructor is done */
1026        EventQueue.invokeLater(new Runnable() {
1027            public void run() {
1028                init();
1029                frame.setVisible(true);
1030            }
1031        });
1032    }
